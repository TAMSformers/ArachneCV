45 void acvCamera::acvWarpPerspective( )
46 {
47  /* point arrays to generate transform matrix */
48  std::vector<cv::Point2f> src_pts( 4 );
49  std::vector<cv::Point2f> dst_pts( 4 );
50 
51  /* initialize source array */
52  cv::Point2f S0( 0.0, 0.0 );
53  cv::Point2f S1( (float)cam_size.width, 0.0 );
54  cv::Point2f S2( (float)cam_size.width, (float)cam_size.height );
55  cv::Point2f S3( 0.0, (float)cam_size.height );
56  src_pts[ 0 ] = S0;
57  src_pts[ 1 ] = S1;
58  src_pts[ 2 ] = S2;
59  src_pts[ 3 ] = S3;
60 
61  /* find height of triangle as a function of angle */
62  int height = cam_size.width * tan( cam_angle * ( PI / 180.0 ) );
63 
64  /* intersect triangle with image */
65  if( height <= cam_size.height ){
66  cv::Point2f D0( cam_size.width / 2, height );
67  cv::Point2f D1( cam_size.width / 2, height );
68  cv::Point2f D2( cam_size.width, cam_size.height );
69  cv::Point2f D3( 0, cam_size.height );
70  dst_pts[ 0 ] = D0;
71  dst_pts[ 1 ] = D1;
72  dst_pts[ 2 ] = D2;
73  dst_pts[ 3 ] = D3;
74 }
75  else{
76  int offset = ( cam_size.height * ( cam_size.width/2 ) ) / height;
77  cv::Point2f D0( offset, height );
78  cv::Point2f D1( cam_size.width - offset, height );
79  cv::Point2f D2( cam_size.width, cam_size.height );
80  cv::Point2f D3( 0, cam_size.height );
81  dst_pts[ 0 ] = D0;
82  dst_pts[ 1 ] = D1;
83  dst_pts[ 2 ] = D2;
84  dst_pts[ 3 ] = D3;
85 }
86 
87  /* generate transform matrix */
88  cv::InputArray M = cv::getPerspectiveTransform( src_pts, dst_pts );
89 
90  /* apply transform matrix */
91  cv::warpPerspective( frame, frame, M, cam_size );
92 }

94 }
